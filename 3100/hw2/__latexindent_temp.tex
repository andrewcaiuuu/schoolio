\section{Using properties of probability laws:}
\begin{enumerate}[label=\alph*.]
  \item \begin{math}
    Pr(C \cup P) = Pr((C' \cap P')' = 40\%
  \end{math}
  \item \begin{math}
    Pr(C \cap P) = Pr(C) + Pr(P) - Pr(C \cup P) =
    20\% + 30 \% - 40 \% = 10\%
  \end{math}
\end{enumerate}
